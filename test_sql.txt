1. To form category counts
SELECT
    t.term_id AS category_id,
    t.name AS category_name,
    COUNT(p.ID) AS post_count
FROM
    xhk_terms AS t
JOIN
    xhk_term_taxonomy AS tt ON t.term_id = tt.term_id
JOIN
    xhk_term_relationships AS tr ON tt.term_taxonomy_id = tr.term_taxonomy_id
JOIN
    xhk_posts AS p ON tr.object_id = p.ID
WHERE
    tt.taxonomy = 'category'
    AND p.post_status = 'publish'
GROUP BY
    t.name
ORDER BY
    post_count DESC;

2. To get id ordered by date in (29 - News, 27 - E&BE; 23 - Act with Art)
SELECT
    distinct p.ID
FROM
    xhk_posts AS p
JOIN
    xhk_term_relationships AS tr ON tr.object_id = p.ID
JOIN
     xhk_term_taxonomy AS tt ON tt.term_taxonomy_id = tr.term_taxonomy_id
WHERE
    tt.taxonomy = 'category'
    AND p.post_status = 'publish' and post_type='post'
    AND tt.term_id in (29,27,23)
ORDER BY
    post_date DESC;

4. To get post id, date, author, categories
SELECT 
    p.ID, 
    p.post_date, 
    p.post_author, 
    GROUP_CONCAT(t.term_id) AS category_ids
FROM 
    xhk_posts p
LEFT JOIN 
    xhk_term_relationships tr ON tr.object_id = p.ID
LEFT JOIN 
    xhk_term_taxonomy t ON t.term_taxonomy_id = tr.term_taxonomy_id AND t.taxonomy = 'category'
WHERE 
    p.post_status = 'publish' 
    AND p.post_type = 'post'
GROUP BY 
    p.ID, p.post_date, p.post_author
ORDER BY 
    p.post_date DESC;
